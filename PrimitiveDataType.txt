class PimitiveDataTypes
{
	public static void main(String args[])
	{
		byte x=2;
		System.out.println("Byte value is :"+x);
		int a=4;
		System.out.println("Int value is :"+a);
		long b=6;
		System.out.println("Long value is :"+b);
		short c=8;
		System.out.println("Short value is :"+c);
		double d=10;
		System.out.println("Double value is :"+d);
		char e=12;
		System.out.println("Char value is :"+e);
	}
}
